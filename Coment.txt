const n = 5; — визначає розмірність масиву (в даному випадку 5 x 5). Це значення можна змінити.
var arr: array[1..n, 1..n] of integer; — оголошення двовимірного масиву для зберігання цілих чисел, що має розміри n x n.
var i, j: integer; — оголошення змінних для індексів циклів: i для рядків і j для стовпців.
var sum: integer; — змінна для зберігання суми елементів головної діагоналі масиву.
var avg: real; — змінна для зберігання середнього арифметичного.
writeln('Введіть елементи масиву розміром ', n, ' x ', n); — виведення повідомлення для користувача.
for i := 1 to n do — початок циклу для проходження по рядках масиву.
for j := 1 to n do — вкладений цикл для проходження по стовпцях масиву.
write('arr[', i, ',', j, ']: '); — виведення запиту для введення елемента масиву.
read(arr[i, j]); — введення елемента масиву на позицію arr[i, j].
sum := 0; — ініціалізація змінної sum значенням 0.
for i := 1 to n do — цикл для проходження по головній діагоналі масиву (елементи з індексами i = j).
sum := sum + arr[i, i]; — додаємо елемент головної діагоналі до суми.
avg := sum / n; — обчислюємо середнє арифметичне елементів головної діагоналі.
writeln('Середнє арифметичне елементів головної діагоналі: ', avg:0:2); — виведення результату з точністю до двох знаків після коми.
